import { Component, EventEmitter, Input, Output } from '@angular/core';
import { Stop, StopDTO } from 'src/app/dtos/tour';

/**
 * Card for displaying a tour stop.
 */
@Component({
  selector: 'app-tour-stop-card',
  templateUrl: './tour-stop-card.component.html',
  styleUrls: ['./tour-stop-card.component.scss']
})
export class StopCardComponent {

  @Input() data!: Stop | StopDTO

  /**
   * Whether the stop is editable (will not actually perform the action, just emit the user interacton)
   * Should be set to true only in the tour edit page.
   */
  @Input() editable: boolean = false

  /**
   * Whether the stop is deletable (will not actually perform the action, just emit the user interacton)
   * Should be set to true only in the tour edit page.
   */
  @Input() deletable: boolean = false

  @Output() editClick: EventEmitter<void> = new EventEmitter<void>()
  @Output() deleteClick: EventEmitter<void> = new EventEmitter<void>()

}

<mat-card>
    <mat-card-header>
        <div class="w-full flex justify-between">
            <div>
                <mat-card-title class="!inline">{{ data.title }}</mat-card-title>
                <mat-card-subtitle class="pl-2 !inline">{{ (data.cost | currency) + " &#x25cf; " + data.duration }}</mat-card-subtitle>
            </div>
            <div>
                <button mat-icon-button *ngIf="editable" (click)="editClick.emit()">
                    <mat-icon>edit</mat-icon>
                </button>
                <button mat-icon-button *ngIf="deletable" (click)="deleteClick.emit()">
                    <mat-icon>delete</mat-icon>
                </button>
            </div>
        </div>
    </mat-card-header>
    <mat-card-content>
        <p>{{ data.description }}</p>
        
        <mat-divider class="!my-3"></mat-divider>

        <div class="!mb-2">
            <h3 class="!mb-0">Come raggiungerla:</h3>
            <p>{{ data.transferDetails }}</p>
        </div>
        <ul class="list-disc list-inside ml-2">
            <li>{{ "Tipologia: " + data.transferType }}</li>
            <li>{{ "Costo: " + (data.transferCost | currency) }}</li>
            <li>{{ "Durata: " + data.transferDuration }}</li>
        </ul>

        <div class="mt-2">
            <h3 class="!mb-0">Altre opzioni:</h3>
            <p class="!mb-0">{{ data.transferOtherOptions }}</p>
        </div>

    </mat-card-content>
</mat-card>
